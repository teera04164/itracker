const SHEET_NAME='Items';
function getSheet(){
  const ss=SpreadsheetApp.getActiveSpreadsheet();
  let s=ss.getSheetByName(SHEET_NAME);
  if(!s){s=ss.insertSheet(SHEET_NAME);
    s.appendRow(['ID','ชื่อของ','ตำแหน่ง','กลุ่ม ID','หมวดหมู่','หมายเหตุ','URL รูป','Workspace','วันที่']);
    s.getRange(1,1,1,9).setFontWeight('bold').setBackground('#f5a623').setFontColor('#fff');
    s.setFrozenRows(1);}return s;}
function doGet(e){
  const a=e.parameter.action||'get',s=getSheet();
  if(a==='get'){
    const r=s.getDataRange().getValues(),items=[];
    for(let i=1;i<r.length;i++){if(!r[i][0])continue;
      items.push({id:String(r[i][0]),name:r[i][1],location:r[i][2],
        groupId:r[i][3],category:r[i][4],note:r[i][5],
        image:r[i][6],workspace:r[i][7],createdAt:r[i][8]});}
    return out({items});}
  if(a==='add'){
    s.appendRow([e.parameter.id,e.parameter.name,e.parameter.location,
      e.parameter.groupId||'',e.parameter.category||'',e.parameter.note||'',
      e.parameter.image||'',e.parameter.workspace||'',e.parameter.createdAt]);
    return out({success:true});}
  if(a==='delete'){
    const r=s.getDataRange().getValues();
    for(let i=1;i<r.length;i++){if(String(r[i][0])===String(e.parameter.id)){s.deleteRow(i+1);break;}}
    return out({success:true});}
  return out({error:'unknown'});}
function out(o){return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON);}
